37  switch (min_level) {
38  case Log_level::Trace: {
39  break;
40  }
41  case Log_level::Debug: {
42  if (level == Log_level::Trace) {
43  result = false;
44  }
45  break;
46  }
47  case Log_level::Info: {

51  }
52  break;
53  }
54  case Log_level::Warning: {
55  if (level == Log_level::Trace ||
56  level == Log_level::Debug ||

58  result = false;

64  level != Log_level::Fatal) {

71  result = false;
72  }

74  }

77  }
